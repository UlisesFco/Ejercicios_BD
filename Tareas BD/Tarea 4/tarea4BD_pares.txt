select nomEmp, sum(aportacion) "suma aportacion", count(*) "Congresos participados" 
from empresa_t4 e, organiza_t4 o, congreso_t4 c
where e.idorg = o.idorg and c.idcon = o.idcon
group by nomEmp
having count(*) >2;
